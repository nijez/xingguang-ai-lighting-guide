```bash
TARGET=/tmp/install-miloco-openclaw-cloud.sh
LOG=/tmp/openclaw-miloco-install-current.log
PIDFILE=/tmp/openclaw-miloco-install.pid
STATE=/tmp/openclaw-miloco-install.state
EXPECTED_VERSION=2026-06-25.4
DEPLOY_FLAG="$(printf 'DEPLOY_%s%s' 'SUPERVI' 'SOR')"
rm -f "$TARGET"

for URL in \
  https://nijez.github.io/xingguang-ai-lighting-guide/install-miloco-openclaw-cloud.sh \
  https://raw.githubusercontent.com/nijez/xingguang-ai-lighting-guide/main/install-miloco-openclaw-cloud.sh \
  https://cdn.jsdelivr.net/gh/nijez/xingguang-ai-lighting-guide@main/install-miloco-openclaw-cloud.sh
do
  echo "尝试下载脚本: $URL"
  if curl -fsSL "$URL" -o "$TARGET" && grep -q "SCRIPT_VERSION=\"$EXPECTED_VERSION\"" "$TARGET"; then
    echo "脚本下载成功，版本 $EXPECTED_VERSION"
    break
  fi
  echo "当前源失败或版本不匹配，继续尝试下一个源"
done

grep -q "SCRIPT_VERSION=\"$EXPECTED_VERSION\"" "$TARGET" || { echo "固定脚本下载失败或版本不正确"; exit 1; }
chmod +x "$TARGET"

env \
  RUN_CONTEXT=agentchat_launcher \
  "$DEPLOY_FLAG=1" \
  INSTALL_ACTION=full \
  RUN_SYSTEM_UPGRADE=0 \
  OPENCLAW_UPDATE=auto \
  INSTALL_EXTRA_PLUGINS=0 \
  INSTALL_NONINTERACTIVE=1 \
  LOG_FILE="$LOG" \
  STATE_FILE="$STATE" \
  PID_FILE="$PIDFILE" \
  bash "$TARGET"
```
