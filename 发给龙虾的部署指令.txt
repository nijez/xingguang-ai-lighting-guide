- 不要用英文写 “I'll / Let me / Good / It's progressing” 这类过程说明。

目标：
1. 检查已预装 OpenClaw 和 gateway 状态。
2. 轻量部署 Miloco 2.0 底座。
3. 默认跳过微信绑定、米家账号绑定、MiMo API Key 配置。

当前阶段只部署底座，不安装馨光智能 Skill。馨光 Skill、设备白名单、淡彩光颜色规则和真机验收属于第二阶段。

请只执行下面这一份固定流程：

```bash
TARGET=/tmp/install-miloco-openclaw-cloud.sh
LOG=/tmp/openclaw-miloco-install-current.log
PIDFILE=/tmp/openclaw-miloco-install.pid
STATE=/tmp/openclaw-miloco-install.state
EXPECTED_VERSION=2026-06-24.2
rm -f "$TARGET"

for URL in \
  https://nijez.github.io/xingguang-ai-lighting-guide/install-miloco-openclaw-cloud.sh \
  https://raw.githubusercontent.com/nijez/xingguang-ai-lighting-guide/main/install-miloco-openclaw-cloud.sh \
  https://cdn.jsdelivr.net/gh/nijez/xingguang-ai-lighting-guide@main/install-miloco-openclaw-cloud.sh
do
  echo "尝试下载脚本: $URL"
  if curl -fsSL "$URL" -o "$TARGET" && grep -q "SCRIPT_VERSION=\"$EXPECTED_VERSION\"" "$TARGET"; then
    echo "脚本下载成功，版本 $EXPECTED_VERSION"
    break
  fi
  echo "当前源失败或版本不匹配，继续尝试下一个源"
done

grep -q "SCRIPT_VERSION=\"$EXPECTED_VERSION\"" "$TARGET" || { echo "固定脚本下载失败或版本不正确"; exit 1; }
chmod +x "$TARGET"

rm -f "$LOG" "$PIDFILE"
nohup bash -lc 'INSTALL_ACTION=full RUN_SYSTEM_UPGRADE=0 OPENCLAW_UPDATE=0 INSTALL_EXTRA_PLUGINS=0 INSTALL_NONINTERACTIVE=1 bash /tmp/install-miloco-openclaw-cloud.sh || INSTALL_ACTION=continue RUN_SYSTEM_UPGRADE=0 OPENCLAW_UPDATE=0 INSTALL_EXTRA_PLUGINS=0 INSTALL_NONINTERACTIVE=1 bash /tmp/install-miloco-openclaw-cloud.sh' > "$LOG" 2>&1 &
echo $! > "$PIDFILE"
PID="$(cat "$PIDFILE")"
START_TS="$(date +%s)"
LAST_SIZE=0
LAST_CHANGE="$START_TS"

echo "安装进程已启动 PID=$PID"
echo "日志文件: $LOG"
echo "状态文件: $STATE"

for i in $(seq 1 90); do
  NOW="$(date +%s)"
  SIZE="$(wc -c < "$LOG" 2>/dev/null || echo 0)"
  if [ "$SIZE" != "$LAST_SIZE" ]; then
    LAST_SIZE="$SIZE"
    LAST_CHANGE="$NOW"
  fi
  IDLE="$((NOW - LAST_CHANGE))"
  ELAPSED="$((NOW - START_TS))"
  STEP="$(grep -E 'Step [0-9]+/6|STATE: STEP_[0-9]+_DONE|ERROR|Failed|Traceback|timeout|Timing:|Done' "$LOG" 2>/dev/null | tail -n 1 || true)"

  echo
  echo "===== 安装进度检查 $i/90 ====="
  if kill -0 "$PID" 2>/dev/null; then
    echo "状态: RUNNING"
  elif [ -f "$STATE" ] && grep -q '^STEP_6_DONE$' "$STATE"; then
    echo "状态: FINISHED"
  else
    echo "状态: EXITED_BUT_INCOMPLETE"
  fi
  echo "已运行: ${ELAPSED}s"
  echo "日志最后更新: ${IDLE}s 前"
  echo "当前进度: ${STEP:-尚未写入步骤日志，可能正在初始化或下载}"

  echo "--- 关键进度 ---"
  grep -E 'Step [0-9]+/6|STATE: STEP_[0-9]+_DONE|ERROR|Failed|Traceback|timeout|Timing:|Done' "$LOG" 2>/dev/null | tail -n 40 || true

  if kill -0 "$PID" 2>/dev/null && [ "$IDLE" -ge 180 ]; then
    echo "--- 无新日志提醒 ---"
    echo "安装进程仍在运行，但暂无新日志。可能正在下载、解压、解析依赖或等待 OpenClaw gateway 响应。"
    ps -p "$PID" -o pid,etime,stat,pcpu,pmem,cmd 2>/dev/null || true
    pgrep -af 'install-miloco|uv|curl|npm|node|openclaw|miloco' 2>/dev/null | head -n 20 || true
  fi

  if ! kill -0 "$PID" 2>/dev/null; then
    break
  fi
  sleep 20
done

if kill -0 "$PID" 2>/dev/null; then
  echo "安装仍在后台运行，稍后可发送查看安装进度指令继续查看。"
fi
```
