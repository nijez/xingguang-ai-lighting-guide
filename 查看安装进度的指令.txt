PIDFILE=/tmp/openclaw-miloco-install.pid
LOG=/tmp/openclaw-miloco-install-current.log
STATE=/tmp/openclaw-miloco-install.state
NOW="$(date +%s)"
CONTINUE_CMD='INSTALL_ACTION=continue RUN_SYSTEM_UPGRADE=0 OPENCLAW_UPDATE=0 INSTALL_EXTRA_PLUGINS=0 INSTALL_NONINTERACTIVE=1 bash /tmp/install-miloco-openclaw-cloud.sh'

if [ -f "$PIDFILE" ] && kill -0 "$(cat "$PIDFILE")" 2>/dev/null; then
  STATUS=RUNNING
elif [ -f "$STATE" ] && grep -q '^STEP_6_DONE$' "$STATE"; then
  STATUS=FINISHED
else
  STATUS=EXITED_BUT_INCOMPLETE
fi
echo "状态: $STATUS"

if [ -f "$STATE" ]; then
  LAST_DONE="$(grep -E '^STEP_[0-9]+_DONE$' "$STATE" | tail -n 1 || true)"
else
  LAST_DONE=""
fi

NEXT_STEP=COMPLETE
for n in 1 2 3 4 5 6; do
  if ! grep -q "^STEP_${n}_DONE$" "$STATE" 2>/dev/null; then
    NEXT_STEP="STEP_${n}"
    break
  fi
done

if [ -f "$LOG" ]; then
  LOG_MTIME="$(stat -c %Y "$LOG" 2>/dev/null || date +%s)"
  IDLE="$((NOW - LOG_MTIME))"
  STEP="$(grep -E 'Step [0-9]+/6|STATE: STEP_[0-9]+_DONE|ERROR|Failed|Traceback|timeout|Timing:|Done' "$LOG" 2>/dev/null | tail -n 1 || true)"
  ERRORS="$(grep -Ei 'ERROR|Failed|Traceback|timeout|timed out' "$LOG" 2>/dev/null | tail -n 20 || true)"
else
  IDLE=-1
  STEP=""
  ERRORS=""
fi

echo "最后完成步骤: ${LAST_DONE:-none}"
echo "下一步: $NEXT_STEP"
echo "当前进度: ${STEP:-尚未找到步骤日志}"
if [ "$IDLE" -ge 0 ]; then
  echo "日志最后更新: ${IDLE}s 前"
fi
echo "日志文件: $LOG"
echo "状态文件: $STATE"

if [ "$STATUS" = RUNNING ] && [ "$IDLE" -ge 180 ]; then
  echo "--- 无新日志提醒 ---"
  echo "安装进程仍在运行，但暂无新日志。可能正在下载、解压、解析依赖或等待 OpenClaw gateway 响应。"
  PID="$(cat "$PIDFILE")"
  ps -p "$PID" -o pid,etime,stat,pcpu,pmem,cmd 2>/dev/null || true
  pgrep -af 'install-miloco|uv|curl|npm|node|openclaw|miloco' 2>/dev/null | head -n 20 || true
fi

if [ "$STATUS" = EXITED_BUT_INCOMPLETE ]; then
  echo "--- 未完成状态 ---"
  echo "安装进程已退出，但状态文件没有 STEP_6_DONE。"
  echo "是否可以继续: 是"
  echo "建议继续命令: $CONTINUE_CMD"
fi

if [ -n "$ERRORS" ]; then
  echo "--- 最近异常关键词 ---"
  printf '%s\n' "$ERRORS"
fi

echo "--- 最近日志 ---"
tail -n 120 "$LOG" 2>/dev/null || echo "暂未找到日志"

echo "--- Miloco service ---"
miloco-cli service status 2>/dev/null || true

echo "--- OpenClaw gateway ---"
openclaw gateway status 2>/dev/null || true

echo "--- Miloco plugin ---"
openclaw plugins list 2>/dev/null | grep -i -C 2 miloco || true
